Port Scanner (Python)

import socket

host = "scanme.nmap.org"
for port in range(1,100):
    s = socket.socket()
    s.settimeout(0.3)
    try:
        s.connect((host, port))
        print(f"Porta {port} aberta")
    except:
        pass
